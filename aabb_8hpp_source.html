318  static bool IntersectObject(const OGLKit::Vector3<T>& p,
319  const OGLKit::Vector3<T>& dir,
320  const AABB<T>& bbox,
321  T* t) {
322  // Check for intersection
323  T t_enter = 0.f;
324  T t_exit = 1.f;
325  // For all three slabs
326  const T* dir_ptr = &dir.x_;
327  const T* p_ptr = &p.x_;
328  const T* bbox_min_ptr = &bbox.min_.x_;
329  const T* bbox_max_ptr = &bbox.max_.x_;
330  for (int i = 0; i < 3; ++i) {
331  if (std::abs(*dir_ptr) < 1e-9) {
332  // Ray is parallel to slab. No hit if origin not within slab
333  if ((*p_ptr < *bbox_min_ptr) || (*p_ptr > *bbox_max_ptr)) {
334  return false;
335  }
336  } else {
337  // Compute intersection t value of ray with near and far plane of slab
338  T ood = 1.0f / *dir_ptr;
339  T t_slab_enter = (*bbox_min_ptr - *p_ptr) * ood;
340  T t_slab_exit = (*bbox_max_ptr - *p_ptr) * ood;
341  // Make t_slab_enter be intersection with near plane, t_slab_exit with
342  // far plane
343  if (t_slab_enter > t_slab_exit) {
344  std::swap(t_slab_enter, t_slab_exit);
345  }
346  // make sure the slab interval and the current box intersection
347  // interval overlap
348  if (t_slab_enter > t_exit || t_slab_exit < t_enter) {
349  // nope. Ray missed the box.
350  return false;
351  } else {
352  // yep, the slab and current intersection interval overlap
353  // update the intersection interval
354  // Compute the intersection of slab intersection intervals
355  if (t_slab_enter > t_enter) { t_enter = t_slab_enter; }
356  if (t_slab_exit < t_exit) { t_exit = t_slab_exit; }
357  // Exit with no collision as soon as slab intersection becomes empty
358  if (t_enter > t_exit) { return false; }
359  }
360  }
361  // Inc ptr
362  ++dir_ptr;
363  ++p_ptr;
364  ++bbox_min_ptr;
365  ++bbox_max_ptr;
366  }
367  // Ray intersects all 3 slabs. Return point (q) and intersection t
368  // value (tmin)
369  *t = t_enter;
370  return true;
371  }
